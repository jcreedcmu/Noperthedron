\begin{theorem}
\label{thm:polyhedron_radius_def}
\lean{polyhedron_radius_def}
\leanok
Suppose $S$ is a finite set of points in $\R^n$. If
\begin{itemize}
\item there is a vector $v \in S$ with $\|v\| = r$
\item all vectors $v \in S$ have $\|v\| \le r$
\end{itemize}
then the radius of the polyhedron $S$ is $r$.
\end{theorem}
\begin{proof}
Immediate from definition.
\end{proof}
